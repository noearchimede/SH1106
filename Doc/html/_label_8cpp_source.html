97 bool Label::print(const char text[], uint16_t length, bool progmem) {
98 
99  // if the string is empty
100  if(!text[0]) return true;
101 
102  uint16_t wordStartIndex = 0;
103  bool charsToPrint = false;
104 
105  uint16_t i = 0;
106  while(i < length) {
107  bool escapeSequence;
108  MoveType move;
109  if(progmem) {
110  move = getMoveChar(escapeSequence, pgm_read_byte(&(text[i])), (i+1 < length? pgm_read_byte(&(text[i+1])) : '\0'));
111  }
112  else {
113  move = getMoveChar(escapeSequence, text[i], (i+1 < length? text[i+1] : '\0'));
114  }
115 
116  if(move != MoveType::none) {
117  // Print last word
118  if(charsToPrint) {
119  charsToPrint = false;
120  if(!writeWord(text, wordStartIndex, i, progmem)) return false;
121  }
122 
123  // Perform required action on cursor
124  switch(move) {
125  case MoveType::space:
126  // avoid spaces at the beginning of a line. This restriction
127  // doesn't apply to the `space()` function.`
128  if(cursor.column != 0)
129  if(!space()) return false;
130  break;
131  case MoveType::tab:
132  if(!tab()) return false;
133  break;
134  case MoveType::newline:
135  if(!newline()) return false;
136  break;
137  case MoveType::carriageReturn:
138  if(!carriageReturnClear()) return false;
139  break;
140  case MoveType::none: break; // Can't happen
141  }
142  }
143  // else we know that the character has to be interpreted as text
144  else {
145  // If this is the first printable character after a space, tab, ...
146  if(!charsToPrint) {
147  wordStartIndex = i;
148  charsToPrint = true;
149  }
150  }
151 
152  // The only possible escapes here are the "action" ones (e.g. "\\t").
153  // The others are analyzed while measuring and printing a word.
154  if(escapeSequence) i++;
155  i++;
156  }
157  // Print last word
158  if(charsToPrint) {
159  charsToPrint = false;
160  if(!writeWord(text, wordStartIndex, i, progmem)) return false;
161  }
162 
163  return true;
164 }

170 bool Label::tab(uint8_t anchor) {
171 
172 
173  if(anchor >= frame.columns) return false;
174 
175  // calculate tab width
176  uint8_t tab;
177  // user-defined anchor
178  if(anchor) {
179  if(cursor.column < anchor) {
180  tab = anchor - cursor.column;
181  }
182  else {
183  // The cursor is already on the right side of the anchor.
184  // Just write a space to make sure that the previous word will not
185  // be joined up with the next one.
186  space();
187  return false;
188  }
189  }
190  // default tab
191  else tab = defaultTab - (cursor.column % defaultTab);
192 
193  // Clear the space jumped by the tab
194  bool inFrame = frame.isInFrame(cursor.column + tab);
195 
196  if(inFrame) fill(0x00, cursor.column, cursor.page, (cursor.column + tab), cursor.page);
197  else fill(0x00, cursor.column, cursor.page, 0xFF, cursor.page);
198 
199  // move the cursor
200  cursor.move(tab, true);
201 
202  if(inFrame) return true;
203  return false;
204 }

248 void Label::fill(uint8_t data, uint8_t beginCol, uint8_t beginPag, uint8_t endCol, uint8_t endPag) {
249 
250  // correct the indexing error created by the margin (see e.g. the
251  // constructor to undestand the "problem")
252  if(beginCol > 0 && beginCol != 0xFF) beginCol += margin;
253  if(endCol > 0 && endCol != 0xFF) endCol += margin;
254 
255  // if one of the end parameters is 0xff replace it with its max value
256  if(endPag == 0xFF) endPag = clearFrame.pages - 1;
257  if(endCol == 0xFF) endCol = clearFrame.columns - 1;
258 
259  // if the range is "negative" return
260  if(endPag < beginPag) return;
261  else if(endPag == beginPag) {
262  if(endCol < beginCol) return;
263  }
264 
265  // If the region to fill is on a single page
266  if(beginPag == endPag) {
267  cursor.page = beginPag;
268  for(int i = beginCol; i <= endCol; i++) {
269  cursor.column = i;
270 
271  driver.writeData(clearFrame.absolutePage(cursor.page), clearFrame.absoluteColumn(cursor.column), data);
272  }
273  }
274 
275  // if multiple pages are to be filled
276  else {
277  // clear (part of) the first line
278  cursor.page = beginPag;
279  for(int i = beginCol; i < clearFrame.columns; i++) {
280  cursor.column = i;
281 
282  driver.writeData(clearFrame.absolutePage(cursor.page), clearFrame.absoluteColumn(cursor.column), data);
283  }
284  // clear intermediate line(s)
285  for(int k = beginPag + 1; k < endPag; k++) {
286  cursor.page = k;
287  for(int i = 0; i < clearFrame.columns; i++) {
288  cursor.column = i;
289  driver.writeData(clearFrame.absolutePage(cursor.page), clearFrame.absoluteColumn(cursor.column), data);
290  }
291  }
292  // clear (part of) the last line
293  cursor.page = endPag;
294  for(unsigned int i = 0; i <= endCol; i++) {
295  cursor.column = i;
296  driver.writeData(clearFrame.absolutePage(cursor.page), clearFrame.absoluteColumn(cursor.column), data);
297  }
298  }
299 
300  cursor.page = beginPag;
301  cursor.column = beginCol;
302 }

373 const uint8_t* Label::getPrintableChar(bool& bothUsed, const char char1, const char char2) {
374 
375 
376  // Default return values, will be changed according to the situation.
377  const uint8_t* c = font.getUnknown();
378  bothUsed = true;
379 
380 
381  // Here begins a series of if - elseif conditions that looks for every
382  // possible escape sequence. The else condition at the end is only reached
383  // if the given text must simply be treated as plain text.
384 
385  // accented vowels
386  if(char2 == '`') {
387  if (char1 == 'a') c = font.getAccentedVowel('a');
388  else if(char1 == 'e') c = font.getAccentedVowel('e');
389  else if(char1 == 'i') c = font.getAccentedVowel('i');
390  else if(char1 == 'o') c = font.getAccentedVowel('o');
391  else if(char1 == 'u') c = font.getAccentedVowel('u');
392  }
393 
394  // Arrows
395  else if(char1 == escape && char2 == '^') c = font.getArrow(0); // \^
396  else if(char1 == '-' && char2 == '^') c = font.getArrow(1); // -^
397  else if(char1 == '-' && char2 == '>') c = font.getArrow(2); // ->
398  else if(char1 == '-' && char2 == 'v') c = font.getArrow(3); // -v
399  else if(char1 == escape && char2 == 'v') c = font.getArrow(4); // \v
400  else if(char1 == 'v' && char2 == '-') c = font.getArrow(5); // v-
401  else if(char1 == '<' && char2 == '-') c = font.getArrow(6); // <-
402  else if(char1 == '^' && char2 == '-') c = font.getArrow(7); // ^-
403 
404  // Digits at exponent position
405  else if(char1 == '^' && ('0' <= char2 && char2 <= '9')) c = font.getExpDigit(char2 - '0');
406 
407  // Special characters
408  // degree symbol
409  else if((char1 == escape || char1 == '^') && char2 == 'o') c = font.getSpecialChar(PageFont::SpecialChar::degree);
410  // Copyright symbol
411  else if(char1 == escape && char2 == 'c') c = font.getSpecialChar(PageFont::SpecialChar::copyright);
412 
413  // Double escape character
414  else if(char1 == escape && char2 == escape) c = font.getAscii(escape);
415 
416  // "Action" characters escaped with the custom escape character are treated
417  // by another function and will be considered as 'unknown' characters.
418  // Thus this function should not be called if the character or character
419  // sequence is a tab, newline or carriage return. Neither should it be
420  // called for spaces, but if it were it would return a fixed width space.
421  else if (char1 == escape && (char2 == 'n' || char2 == 't' || char2 == 'r')) c = font.getUnknown();
422 
423  // Unused escape character. It is ignored as may be used to avoid another
424  // escape sequence: e.g. "->" prints a right arrow, "-\>" prints "->",
425  // where '\\' is the escape character.
426  // There must always be at least two normal characters between two escapes
427  // so that here we can assume that the second character is plain ASCII.
428  else if(char1 == escape) c = font.getAscii(char2);
429 
430 
431  // ### NO SPECIAL CHARACTER SEQUENCE ### //
432 
433  else {
434  // plain ASCII text (will be a real character from ASCII '!' to '~',
435  // i.e. 0x20 to 0x7E, and a printable 'unknown' character otherwise)
436  // with one single exception: the vertical tab will be interpreted
437  // as a space as wide as a character (it is used to align numbers).
438  if (char1 == 0x0B) c = font.getSpecialChar(PageFont::SpecialChar::largeSpace);
439 
440  else c = font.getAscii(char1);
441  // Update default return value
442  bothUsed = false;
443  }
444 
445  return c;
446 }

472 uint16_t Label::getWordWidth(const char word[], uint16_t firstIndex, uint16_t stopIndex, bool progmem) {
473  // All characters are assumed to be printable
474  uint16_t width = 0;
475  uint16_t i = firstIndex;
476  while(i < stopIndex) {
477 
478  // get the character that should be printed
479  const uint8_t* c;
480  bool escapeSequence;
481  if(progmem) {
482  c = getPrintableChar(escapeSequence, pgm_read_byte(&(word[i])), (i < stopIndex ? pgm_read_byte(&(word[i+1])) : '\0'));
483  }
484  else {
485  c = getPrintableChar(escapeSequence, word[i], (i < stopIndex ? word[i+1] : '\0'));
486  }
487 
488  // get its width
489  uint8_t length;
490  font.readCharLenght(c, length);
491  width += length;
492  // Add the one byte space between letters
493  width += 1;
494  // Increase `i` by one or two if two characters were joined together
495  if(escapeSequence) i++;
496  i++;
497  }
498  return width;
499 }

504 bool Label::writeWord(const char word[], uint16_t firstIndex, uint16_t stopIndex, bool progmem) {
505  // All characters passed to this function are assumed to be printable.
506  // If some non printable characters are fond they will be print as
507  // "unknown"
508 
509  // get word width
510  const uint16_t width = getWordWidth(word, firstIndex, stopIndex, progmem);
511 
512  // check whether it is possible to print it and move the cursor if needed
513  if(!cursor.prepare(width)) return false;
514 
515 
516  uint16_t i = firstIndex;
517  while(i < stopIndex) {
518  // Get the printable character
519  const uint8_t* c;
520  bool escapeSequence;
521  if(progmem) {
522  c = getPrintableChar(escapeSequence, pgm_read_byte(&(word[i])), (i < stopIndex ? pgm_read_byte(&(word[i+1])) : '\0'));
523  }
524  else {
525  c = getPrintableChar(escapeSequence, word[i], (i < stopIndex ? word[i+1] : '\0'));
526  }
527  if(!writeChar(c)) return false;
528 
529  // Go to next character
530  if(escapeSequence) i++;
531  i++;
532  }
533  return true;
534 }

565 bool Label::printInt(unsigned long n, uint8_t minWidth, uint8_t base, bool negative) {
566  // This function was taken from the Print class in the Arduino framework
567  // and sligtly modified
568 
569  // dimension: 8*4 binary digits, minus sign/hex symbol, null character
570  char buf[8 * sizeof(long) + 2 + 1];
571  char *str = &buf[sizeof(buf) - 1];
572 
573  *str = '\0';
574 
575  // prevent crash if called with base == 1
576  if (base < 2) base = 10;
577 
578  do {
579  char c = n % base;
580  n /= base;
581  *--str = c < 10 ? c + '0' : c + 'A' - 10;
582  } while(n);
583 
584  // add special symbols (- for negative numbers or 0x for exadecimal base)
585  if(negative) *--str = '-';
586  else if (base == 16) {
587  *--str = 'x';
588  *--str = '0';
589  }
590 
591  // if needed add some spaces to reach the minimum width
592  uint8_t width = &buf[sizeof(buf) - 1] - str;
593  int8_t spaces = minWidth - width;
594  // 0x0B is ascii vertical tab, interpredet by this class as a space as wide
595  // as one digit
596  if(spaces > 0) while(spaces--) *--str = 0x0B;
597 
598  return print(str);
599 }

602 bool Label::printFloat(double n, uint8_t fractDigits, uint8_t minIntDigits) {
603  // Some parts of this functions were copied from the Print class in the
604  // Arduino framework
605 
606  // dimension: minus sign, 10 digits in integral part, dot, 8 digits in
607  // decimal part, null character, (arbitrarly) 4 spaces for alignment
608  char buf[25];
609  char *str = &buf[sizeof(buf) - 1];
610 
611  char *dotAddr = str; //used later for alignment
612 
613  *--str = '\0';
614 
615  // Handle invalid numbers
616  // The two numbers for overflow are marked as "determined empirically" in
617  // the Arduino Print library implementation
618  if (isnan(n)) {*--str = 'n';*--str = 'a';*--str = 'n';}
619  else if (isinf(n)) {*--str = 'f';*--str = 'n';*--str = 'i';}
620  else if (n > 4294967040.0) {*--str = 'f';*--str = 'o';*--str = 'v';}
621  else if (n <- 4294967040.0) {*--str = 'f';*--str = 'o';*--str = 'v';}
622 
623  // N is a valid number
624  else {
625 
626  // Round correctly so that print(1.999, 2) prints as "2.00"
627  double rounding = 0.5;
628  for (uint8_t i = 0; i < fractDigits; ++i) rounding /= 10.0;
629  n += rounding;
630 
631  // Extract the integer part of the number
632  unsigned long intPart = (unsigned long)n;
633  double rem = n - (double)intPart;
634 
635  // Extract digits from the remainder
636  uint8_t actualFractDigits = 0;
637  for(;actualFractDigits < fractDigits; actualFractDigits++) rem *= 10;
638  // write in string
639  do {
640  char c = (unsigned long)rem % 10;
641  rem /= 10;
642  *--str = c + '0';
643  } while(--actualFractDigits);
644 
645  // Print the decimal point, but only if there are digits beyond
646  if (fractDigits > 0) {
647  dotAddr = str;
648  *--str = '.';
649  }
650 
651  // count digits in intPart
652  uint8_t intDigits = 0;
653  for(uint8_t x = intPart; x /= 10; intDigits++);
654  // write in string
655  do {
656  char c = intPart % 10;
657  intPart /= 10;
658  *--str = c + '0';
659  } while(intPart);
660 
661  // Handle negative numbers
662  if (n < 0.0)
663  {
664  *--str = '-';
665  n = -n;
666  }
667 
668  }
669 
670  // if needed add some spaces to reach the minimum width
671  uint8_t width = dotAddr - str;
672  int8_t spaces = minIntDigits - width + 1;
673  // 0x0B is ascii vertical tab, interpredet by this class as a space as wide
674  // as one digit
675  if(spaces > 0) while(spaces--) *--str = 0x0B;
676  // additional spaces if the number will be printed as a three letter code
677  if(isnan(n) || isinf(n) || n > 4294967040.0 || n <- 4294967040.0)
678  for(uint8_t i = 0; i < fractDigits; i++) *--str = 0x0B;
679 
680  return print(str);
681 }

735 bool Label::Cursor::prepare(uint8_t textWidth) {
736 
737  // The cursor is already out of frame
738  if(page >= frame.pages || column >= frame.columns) return false;
739 
740  // there is enough space starting from current location, do nothing
741  if(column + textWidth < frame.columns) return true;
742 
743  bool availableLine;
744  if(page + 1 < frame.pages) availableLine = true;
745  else if(frame.infinite) availableLine = true;
746  else availableLine = false;
747 
748  // there isn't, try to move to next line but before doing it check if the
749  // word would fit in an empty line.
750  if(textWidth > frame.columns && availableLine) {
751  // the word is too long to be written on a single page, but there are
752  // enough pages to write it. Don't move to next line.
753  return true;
754  }
755  if(availableLine) {
756  // ok, there is another available line
757  move(0xff, true);
758  return true;
759  }
760  // no available page, don't move and return false
761  return false;
762 }

766 bool Label::Cursor::move(uint8_t steps, bool stopAtNL) {
767  // if needed go to another line
768  while(column + steps >= frame.columns) {
769  steps -= frame.columns - column;
770  if(frame.infiniteEmptyLine) {
771  uint8_t p = page;
772  label.fill(0x00, 0, p + 2, 0xFF, p + 2);
773  label.fill(0x00, 0, p + 1, 0xFF, p + 1);
774  page = p + 1;
775  }
776  else page++;
777  column = 0;
778  // the cursor went past the last line
779  if(page >= frame.pages) {
780  if(frame.infinite) {
781  if(frame.infiniteEmptyLine) label.fill(0x00, 0, 1, 0xFF, 1);
782  page = 0;
783  }
784  else return false;
785  }
786  // stopAtNL option
787  if(stopAtNL) {
788  return true;
789  }
790  }
791  // now column + steps < totCol
792  column += steps;
793  return true;
794 }
